我来设计一个**完整优化后的课件文件结构**，从底层到上层全方位优化：

## 一、整体文件布局

```
COURSE.CW (自定义课件文件)
├── [头部区域]      (128字节) - 文件全局信息
├── [索引区域]      (变长)    - 快速查找表
├── [字典区域]      (变长)    - 字符串/样式字典
├── [数据区域]      (变长)    - 实际内容数据
├── [资源池]        (变长)    - 多媒体资源
├── [缓存区域]      (变长)    - 预计算数据
└── [尾部区域]      (32字节)  - 校验和/签名
```

## 二、详细结构定义

### 1. **头部区域 (Header) - 128字节**

```c
#pragma pack(push, 1)  // 1字节对齐，避免填充

typedef struct {
    // 文件标识 (16字节)
    char     magic[8];           // "COURSEW1" - 8字节标识+版本
    uint64_t file_uuid;           // 文件唯一ID
    
    // 版本信息 (8字节)
    uint16_t major_version;       // 主版本
    uint16_t minor_version;       // 次版本
    uint32_t format_flags;        // 格式标志
    
    // 偏移量表 (40字节)
    uint64_t index_offset;        // 索引区偏移
    uint64_t dict_offset;         // 字典区偏移
    uint64_t data_offset;         // 数据区偏移
    uint64_t asset_offset;        // 资源区偏移
    uint64_t cache_offset;        // 缓存区偏移
    
    // 大小信息 (32字节)
    uint64_t index_size;          // 索引区大小
    uint64_t dict_size;           // 字典区大小
    uint64_t data_size;           // 数据区大小
    uint64_t asset_size;          // 资源区大小
    
    // 压缩/加密信息 (16字节)
    uint8_t  compress_method;      // 0:无,1:zstd,2:lz4,3:gzip
    uint8_t  encrypt_method;       // 0:无,1:AES-128,2:AES-256
    uint8_t  block_size_pow;       // 压缩块大小(2的幂)
    uint8_t  padding[13];          // 填充对齐
    
    // 校验信息 (16字节)
    uint32_t header_crc32;         // 头部校验
    uint64_t total_size;           // 文件总大小
    uint32_t file_crc32;           // 文件校验
} CourseHeader;

#pragma pack(pop)
```

### 2. **索引区域 (Index) - B+树结构**

```c
// 索引节点 (通用结构)
typedef struct {
    uint8_t  node_type;            // 0:叶子,1:内部节点
    uint16_t key_count;            // 键数量
    uint32_t block_size;           // 本块大小
    
    union {
        // 内部节点
        struct {
            uint64_t keys[32];      // 键值（幻灯片ID）
            uint64_t children[33];  // 子节点偏移
        } internal;
        
        // 叶子节点
        struct {
            uint64_t keys[64];      // 键值（幻灯片ID）
            struct {
                uint64_t offset;    // 数据偏移
                uint32_t size;      // 数据大小
                uint16_t flags;     // 属性标志
                uint16_t ref_count; // 引用计数
            } values[64];
        } leaf;
    } data;
    
    uint64_t next_leaf;            // 下一个叶子节点(仅叶子有效)
    uint32_t node_crc;             // 节点校验
} IndexNode;

// 快速访问表 (位于索引区开头)
typedef struct {
    uint32_t slide_count;           // 幻灯片总数
    uint32_t master_count;          // 母版数量
    uint32_t layout_count;          // 布局数量
    uint64_t root_index;            // B+树根节点偏移
    uint64_t free_list;             // 空闲块链表
    uint64_t last_slide_id;         // 最后使用的ID
} IndexHeader;
```

## 四、压缩策略优化

```c
// 智能压缩选择
CompressionMethod selectCompression(void* data, size_t size) {
    if (size < 1024) {
        return COMPRESS_NONE;  // 小数据不压缩
    }
    
    // 分析数据熵值
    double entropy = calculateEntropy(data, size);
    
    if (entropy > 0.8) {
        return COMPRESS_ZSTD;  // 高熵数据用ZSTD
    } else if (entropy > 0.5) {
        return COMPRESS_LZ4;   // 中熵数据用LZ4(更快)
    } else {
        // 检查重复模式
        if (hasRepeatedPatterns(data, size)) {
            return COMPRESS_RLE; // 重复模式用RLE
        }
        return COMPRESS_NONE;
    }
}
```

## 六、文件格式验证器

```javascript
class CoursewareValidator {
    async validate(file) {
        const header = await this.readHeader(file);
        
        // 1. 魔数验证
        if (header.magic !== 'COURSEW1') {
            throw new Error('Invalid file format');
        }
        
        // 2. 版本兼容性检查
        if (header.major_version > SUPPORTED_VERSION) {
            throw new Error('File version too new');
        }
        
        // 3. 完整性检查
        const crc = await this.calculateCRC(file);
        if (crc !== header.file_crc32) {
            throw new Error('File corrupted');
        }
        
        // 4. 索引验证
        await this.verifyIndex(file, header);
        
        // 5. 资源验证
        await this.verifyAssets(file, header);
        
        return true;
    }
}
```
